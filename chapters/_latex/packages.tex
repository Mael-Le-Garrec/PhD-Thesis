\newcommand{\chaptertoc}[1][Contents]{%

  \setlength{\cftsecindent}{0.2cm}
  \setlength{\cftsubsecindent}{0.8cm}
  \setlength{\cftsubsubsecindent}{1.4cm}

  % Reduce the spacing between the list items
  \begin{spacing}{0.1}
    \localtableofcontents%
  \end{spacing}
}
