\newglossaryentry{equipment:LBDS}{type=equipment,name=LBDS, description={LHC Beam Dump System }}
\newglossaryentry{equipment:Crosstalk}{type=equipment,name=Crosstalk, description={Interferences between two electronic circuits }}
\newglossaryentry{equipment:Laundau Octupole}{type=equipment,name=Laundau Octupole, description={Octupoles that introduce a spread in the beam, making it more stable }}
\newglossaryentry{equipment:BPM}{type=equipment,name=BPM, description={ Beam Position Monitor, gives the transverse position of the beam }}
\newglossaryentry{equipment:DOROS}{type=equipment,name=DOROS, description={ Low noise BPM. Currently can't be used with other BPMs due to synchronization issues  }}
\newglossaryentry{beam:Dispersion}{type=beam,name=Dispersion, description={ Change of orbit with momentum offset, mainly in the horizontal plane, created by the dipoles}}
\newglossaryentry{beam:Coupling}{type=beam,name=Coupling, description={ Dependance of the transversal planes between each other }}
\newglossaryentry{beam:Emittance}{type=beam,name=Emittance, description={ (\ensuremath{\epsilon}) Unit describing the beam in phase space. A low emittance indicates a beam with a small momentum offset and confined to a small distance }}
\newglossaryentry{beam:beta-function}{type=beam,name=beta-function, description={ Function dependent on $s$, related to the beam transverse size: $$\sigma(s)= \sqrt{\epsilon \cdot \beta(s)}$$ }}
\newglossaryentry{beam:Chromaticity}{type=beam,name=Chromaticity, description={ Tune change with momentum offset. Usually denoted as three orders: $Q'$, $Q''$ and $Q'''$ }}
\newglossaryentry{beam:Aperture}{type=beam,name=Aperture, description={ Maximum physical transverse size the beam can take in the accelerator without suffering losses }}
\newglossaryentry{beam:Dynamic Aperture}{type=beam,name=Dynamic Aperture, description={ Maximum stable aperture. Above that size, the particles become unstable and become lost }}
\newglossaryentry{beam:Waist}{type=beam,name=Waist, description={ Location where the $\beta$-function is at is minimum in an IP. $\beta^*$ refers to $\beta_{waist}$ }}
\newglossaryentry{beam:Waist Shift}{type=beam,name=Waist Shift, description={ Changing the waist to have $\beta^* = \beta_{IP}$ }}
\newglossaryentry{beam:Rigid Waist Shift}{type=beam,name=Rigid Waist Shift, description={ Doing a waist shift by powering all the triplets at once. No individual trim }}
\newglossaryentry{beam:Orbit Feedback}{type=beam,name=Orbit Feedback, description={ System responsible for acquisition and correction of the orbit }}
\newglossaryentry{beam:ATS Factor}{type=beam,name=ATS Factor, description={ Equivalent to the ratio of the virgin $\beta$-function to the $\beta$-function used in the current ATS scheme, at the edge of the arc  }}
